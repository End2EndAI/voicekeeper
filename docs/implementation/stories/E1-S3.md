## Technical Notes

### Edge Function Structure
```typescript
// supabase/functions/process-recording/index.ts
import { serve } from 'https://deno.land/std@0.168.0/http/server.ts';
import { createClient } from 'https://esm.sh/@supabase/supabase-js@2';

serve(async (req) => {
  // CORS headers
  if (req.method === 'OPTIONS') {
    return new Response('ok', { headers: corsHeaders });
  }

  try {
    // Verify auth
    const authHeader = req.headers.get('Authorization');
    if (!authHeader) {
      return new Response(JSON.stringify({ error: 'Missing authorization' }), {
        status: 401
      });
    }

    // Parse multipart form data
    const formData = await req.formData();
    const audioFile = formData.get('audio');
    const formatType = formData.get('format_type');

    // Validate
    if (!audioFile || !formatType) {
      return new Response(JSON.stringify({ error: 'Missing audio or format_type' }), {
        status: 400
      });
    }

    // Mock response (to be replaced in E4-S1 and E4-S2)
    return new Response(JSON.stringify({
      transcription: 'Mock transcription of audio',
      formatted_text: 'Mock formatted text',
      title: 'Mock Title'
    }), {
      headers: { 'Content-Type': 'application/json', ...corsHeaders }
    });
  } catch (error) {
    return new Response(JSON.stringify({ error: error.message }), {
      status: 500
    });
  }
});
```
