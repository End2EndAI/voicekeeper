## Technical Notes

### Processing Service
```typescript
// services/processing.ts
import { supabase } from './supabase';

export interface ProcessingResult {
  transcription: string;
  formatted_text: string;
  title: string;
}

export const processRecording = async (
  audioUri: string,
  formatType: string
): Promise<ProcessingResult> => {
  const { data: { session } } = await supabase.auth.getSession();
  if (!session) throw new Error('Not authenticated');

  // Read audio file
  const response = await fetch(audioUri);
  const blob = await response.blob();

  // Build form data
  const formData = new FormData();
  formData.append('audio', blob, 'recording.m4a');
  formData.append('format_type', formatType);

  // Call edge function
  const result = await fetch(
    `${process.env.EXPO_PUBLIC_SUPABASE_URL}/functions/v1/process-recording`,
    {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${session.access_token}`,
      },
      body: formData,
    }
  );

  if (!result.ok) {
    const error = await result.json().catch(() => ({ error: 'Unknown error' }));
    throw new Error(error.error || `Processing failed: ${result.status}`);
  }

  return await result.json();
};
```

### Web vs Mobile File Handling
On mobile, `audioUri` is a local file path. On web, it may be a blob URL. The `fetch(audioUri)` pattern works for both.
