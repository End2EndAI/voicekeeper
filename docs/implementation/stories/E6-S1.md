## Description

## Technical Notes

### Preferences Service
```typescript
// services/preferences.ts
import { supabase } from './supabase';
import { FormatType } from '../types';

export const getPreferences = async (): Promise<{ default_format: FormatType }> => {
  const { data, error } = await supabase
    .from('user_preferences')
    .select('default_format')
    .single();
  if (error) return { default_format: 'bullet_list' };
  return data;
};

export const updateDefaultFormat = async (format: FormatType): Promise<void> => {
  const { error } = await supabase
    .from('user_preferences')
    .update({ default_format: format })
    .eq('user_id', (await supabase.auth.getUser()).data.user?.id);
  if (error) throw error;
};
```

### PreferencesContext
```typescript
// contexts/PreferencesContext.tsx
interface PreferencesContextType {
```
